## เฉลย

??? note "โค้ดเฉลย"
    ```c
    #include <stdio.h>
    #include <string.h>
    int main() {
        char s[100];
        scanf("%s", s);
        int n = strlen(s);
        for (int i = 0; i < n; i++) {
            if(i % 3 == 2) printf("..*.");
            else printf("..#.");
        }
        printf(".\n");
        // บรรทัดที่ 2
        for (int i = 0; i < n; i++) {
            if(i % 3 == 2) printf(".*.*");
            else printf(".#.#");
        }
        printf(".\n");
        // บรรทัดที่ 3
        for (int i = 0; i < n; i++) {
            if (i != 0 && i % 3 != 1)
                printf("*.%c.", s[i]);
            else
                printf("#.%c.", s[i]);
        }
        if(n % 3 == 0) printf("*\n");
        else printf("#\n");
        // บรรทัดที่ 4
        for (int i = 0; i < n; i++) {
            if(i % 3 == 2) printf(".*.*");
            else printf(".#.#");
        }
        printf(".\n");
        // บรรทัดที่ 5
        for (int i = 0; i < n; i++) {
            if(i % 3 == 2) printf("..*.");
            else printf("..#.");
        }
        printf(".\n");
    }
    ```
