## เฉลย
??? note "เฉลย"
    ```c
    #include <stdio.h>

    int main() {
        char t[10000];
        int s = 0, b = 0;

        scanf("%s", t);

        for (int i = 0; i < 10000; i++){
            if (t[i] == 0) {
                break;
            } else if (t[i] >= 'A' && t[i] <= 'Z') {
                b = 1;
            } else {
                s = 1;
            }
        }

        if (s == 1 && b == 1) printf("Mix");
        if (s == 1 && b == 0) printf("All Small Letter");
        if (s == 0 && b == 1) printf("All Capital Letter");
    }
    ```
